Q1) DELETE ALL THE EMPLOYEES OF A DEPEARTMENT ENTERED BY THE USER. BEFORE DELETING VERIFY THE EXISTENCE OF THE DEPARTMENT IN THE DEPARTMENT TABLE. ALSO DISPLAY THE TOTAL NUMBER OF EMPLOYEES BEING DELETED IF THE DEPARTMENT IS PRESENT OTHERWISE GENERATE AN ERROR MESSAGE 

 ANSWER :
____________
CLEAR SCREEN
SET SERVEROUTPUT ON

DECLARE
    DEPTNO DEPT.DNO%TYPE;
BEGIN
    DEPTNO := &DEPARTMENT_NUMBER;

    -- Verify department exists
    SELECT DNO INTO DEPTNO
    FROM DEPT
    WHERE DNO = DEPTNO;

    -- Delete employees of that department
    DELETE FROM EMP
    WHERE DNO = DEPTNO;
   IF SQL%FOUNND THEN
	DBMS_OUTPUT.PUT_LINE(' NUMBER OF RECORDS DELETED'|| SQL%ROWCOUNT);
   ELSE
	DBMS_OUTPUT.PUT_LINE(' NO EMPLOYEE WORKING IN THE DEPARTMENT'|| DEPTNO);
   END IF ;
 
EXCEPTION
    WHEN NO_DATA_FOUND THEN
        DBMS_OUTPUT.PUT_LINE('DEPARTMENT '||DEPTNO||' NOT PRESENT');
END;
/
