CREATE TABLE FLIGHT_REGISTRATION (
    FLIGHT_NO VARCHAR2(10) ,
    FLIGHT_NAME VARCHAR2(50),
    COMPANY VARCHAR2(50),
    SOURCE VARCHAR2(50),
    DESTINATION VARCHAR2(50),
    DATE_OF_DEPARTURE DATE ,
    DATE_OF_ARRIVAL DATE,
    CAPACITY NUMBER(3),
    FARE NUMBER(10,2),
    NO_OF_CREW_MEMBERS NUMBER(2)
);
